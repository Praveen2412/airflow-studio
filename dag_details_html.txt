          currentDagRunId = dagRunId;

          vscode.postMessage({ command: 'viewLogs', dagRunId, taskId, tryNumber });
        }

          document.getElementById('stateModalTitle').textContent = 'Set Task State';

          document.getElementById('stateModalTitle').textContent = 'Set DAG Run State';
          document.getElementById('stateType').value = 'dagrun';
          document.getElementById('stateDagRunId').value = dagRunId;
          document.getElementById('stateModal').classList.add('active');
        }

          }

        function closeStateModal() {

        }
        
        function closeTriggerModal() {
          document.getElementById('triggerModal').classList.remove('active');

          vscode.postMessage({ command: 'trigger', conf });

          vscode.postMessage({ command: 'viewSource' });

  private escapeHtml(text: string): string {
    const map: Record<string, string> = { '&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;', "'": '&#039;' };
    return text.replace(/[&<>"']/g, m => map[m]);
  }

  private getErrorHtml(message: string): string {
    return `<!DOCTYPE html><html><body><h1>Error</h1><p>${this.escapeHtml(message)}</p></body></html>`;
  }

  private dispose() {
    DagDetailsPanel.panels.delete(this.dagId);
    this.panel.dispose();
    while (this.disposables.length) {
      this.disposables.pop()?.dispose();
    }
  }
